A new window will appear, change the table name to .Measures and click load

Measures :


TotalSales = SUM('US EF Car Sales'[Quantity Sold])

 

Qty Per Month = DIVIDE([TotalSales],DISTINCTCOUNT('Calendar'[Date].[Month]),0)+0


Qty MoM =

VAR LMSales =

    CALCULATE([TotalSales], DATEADD('Calendar'[Date],-1,MONTH))

RETURN

   IF( ISBLANK(LMSales),0, DIVIDE(([TotalSales]-LMSales), LMSales))

 

MOMIndicator =

SWITCH(

    TRUE(), [Qty MoM]>0,

    UNICHAR(9650), UNICHAR(9660))

 

Qty YoY =

VAR PYSales =

    CALCULATE([TotalSales], DATEADD('Calendar'[Date],-1,YEAR))

RETURN

   IF( ISBLANK(PYSales),0, DIVIDE(([TotalSales]-PYSales), PYSales))

 

YOYIndicator =

SWITCH(

    TRUE(), [Qty YoY]>0,

    UNICHAR(9650), UNICHAR(9660))
